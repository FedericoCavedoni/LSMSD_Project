Query molto utilizzate:

.Controllo se esiste review di stesso utente

indice: db.Review.createIndex({ username: 1, game: 1 })

query senza indice: db.Review.find({
  username: "Torsten",
  game: "Pandemic"
})


query con indice: db.Review.find({
  username: "Torsten",
  game: "Pandemic"
})
.hint({ username: 1, game: 1 }).explain("executionStats")


.Controllare che un gioco sia nella libreria di un utente 

indice: db.User.createIndex({ username: 1, library: 1 })

query senza indice: db.User.find({
  username: "torguem",
  library: { $in: ["Caylus"] }
}).explain("executionStats")

query con indice: db.User.find({
  username: "torguem",
  library: { $in: ["Caylus"] }
}).hint({ username: 1, library: 1 }).explain("executionStats")


.Mostrare il rating di un gioco

indice: db.BoardGames.createIndex({ name: 1 })

query senza indice: db.BoardGames.find(
  { name: "Pandemic" },
  { _id: 0, name: 1, rating: 1 }
).explain("executionStats")


query con indice: db.BoardGames.find(
  { name: "Pandemic" },
  { _id: 0, name: 1, rating: 1 }
).hint({ name: 1 }).explain("executionStats")
